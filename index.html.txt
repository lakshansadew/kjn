        // Main generation function
        async function generateVideo() {
            if (isGenerating) return;
            
            const prompt = document.getElementById('prompt-input').value;
            if (!prompt.trim()) {
                alert('Please enter a prompt first');
                return;
            }

            isGenerating = true;
            generationProgress = 0;
            
            // UI transitions
            document.getElementById('hero-section').style.display = 'none';
            document.getElementById('create-section').style.display = 'none';
            document.getElementById('features-section').style.display = 'none';
            document.getElementById('processing-section').classList.remove('hidden');
            
            // Start generation pipeline
            await runGenerationPipeline(prompt);
        }

        async function runGenerationPipeline(prompt) {
            const stages = [
                { id: 'script', name: 'Expanding Cinematic Script', desc: 'AI analyzing prompt structure and emotional arcs...', duration: 3000 },
                { id: 'visual', name: 'Generating Visual Scenes', desc: 'Creating 4K cinematic shots with optimal angles...', duration: 4000 },
                { id: 'audio', name: 'Synthesizing Audio Layers', desc: 'Generating voiceover, music, and sound effects...', duration: 3000 },
                { id: 'sync', name: 'Synchronizing Timeline', desc: 'Matching audio-visual elements and transitions...', duration: 2000 },
                { id: 'render', name: 'Final Rendering', desc: 'Compositing final 1080p HD video output...', duration: 3000 }
            ];

            const totalStages = stages.length;
            
            for (let i = 0; i < totalStages; i++) {
                const stage = stages[i];
                updateStage(stage);
                
                // Activate current log panel
                document.querySelectorAll('[id^="log-"]').forEach(el => el.style.opacity = '0.5');
                if (stage.id !== 'sync' && stage.id !== 'render') {
                    document.getElementById(`log-${stage.id}`).style.opacity = '1';
                    document.getElementById(`log-${stage.id}`).classList.add('neon-border');
                }
                
                // Simulate progress within stage
                const stageProgress = (i / totalStages) * 100;
                const nextStageProgress = ((i + 1) / totalStages) * 100;
                
                await animateProgress(stageProgress, nextStageProgress, stage.duration);
                
                // Special handling for specific stages
                if (stage.id === 'script') await simulateScriptExpansion(prompt);
                if (stage.id === 'visual') await generateScenePreviews();
                if (stage.id === 'audio') initializeAudioVisualizer();
            }
            
            // Complete
            updateProgress(100);
            setTimeout(() => showFinalResult(prompt), 1000);
        }

        function updateStage(stage) {
            document.getElementById('current-stage').textContent = stage.name;
            document.getElementById('stage-description').textContent = stage.desc;
        }

        async function animateProgress(start, end, duration) {
            const steps = 20;
            const increment = (end - start) / steps;
            const stepDuration = duration / steps;
            
            for (let i = 0; i <= steps; i++) {
                const current = start + (increment * i);
                updateProgress(current);
                await sleep(stepDuration);
            }
        }

        function updateProgress(percent) {
            const circle = document.getElementById('progress-circle');
            const text = document.getElementById('progress-text');
            const radius = 52;
            const circumference = 2 * Math.PI * radius;
            const offset = circumference - (percent / 100) * circumference;
            
            circle.style.strokeDashoffset = offset;
            text.textContent = Math.round(percent) + '%';
            generationProgress = percent;
        }

        async function simulateScriptExpansion(prompt) {

            const enhancedScript = document.getElementById('enhanced-script');
            
            // Generate expanded script based on prompt
            const expanded = generateExpandedScript(prompt);
            
            // Typewriter effect
            scriptContainer.innerHTML = '';

                scriptContainer.innerHTML += expanded[i];
                await sleep(30);
            }
            scriptContainer.innerHTML += '...';
            
            // Show full script
            enhancedScript.innerHTML = formatScript(expanded);
        }

        function formatScript(script) {
            return script.split('\n').map(line => {
                if (line.startsWith('දර්ශන') || line.startsWith('SCENE')) {
                    return `<div class="text-neon-cyan font-bold mt-4 font-orbitron">${line}</div>`;
                } else if (line.includes('[') && line.includes(']')) {
                    return `<div class="text-neon-purple ml-4 italic">${line}</div>`;
                } else if (line.startsWith('-')) {
                    return `<div class="text-gray-400 ml-8 text-sm">• ${line.substring(2)}</div>`;
                } else {
                    return `<div class="text-gray-300 ml-4">${line}</div>`;
                }
            }).join('');
        }
